부서번호별 급여의 합, 평균을 출력
	SELECT deptno AS 부서번호, SUM(sal) AS 합, AVG(sal) AS 급여평균 FROM emp GROUP BY deptno;

업무별 급여의 함, 업무명을 출력
	SELECT job AS 업무명, SUM(sal) AS 급여합 FROM emp GROUP BY job;

부서번호별 급여의 평균, 부서번호을 출력
	SELECT deptno AS 부서번호, AVG(sal) AS 급여평균 FROM emp GROUP BY deptno;

ADAMS의 관리자가 소속된 부서의 급여합을 출력

	SELECT deptno FROM emp WHERE empno = (

		)
	)
